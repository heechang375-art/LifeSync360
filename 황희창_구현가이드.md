### 서비스 흐름 (아키텍처 Slide 24~26 기준)

```
고객 브라우저
  → /register  (회원가입)
  → /login     (로그인 → JWT 발급 → localStorage 저장)
  → /consent   (동의 없으면 강제 이동)
  → /          (메인 대시보드)
  → /product/<id>  (상품 상세)
  → /settings  (등급/동의/로그아웃)
```

### JWT 인증 흐름

```
로그인 성공
  → 서버: JWT 생성 { sub: ls_user_id, gid: global_id, exp: 24h }
  → 클라이언트: localStorage.setItem('ls_token', token)

이후 모든 API 요청
  → Header: Authorization: Bearer <token>
  → 서버: JWT decode → sub(ls_user_id), gid(global_id) 추출
  → 유효하지 않으면 401 반환 → 클라이언트가 /login으로 리다이렉트
```

### 대시보드 실제 데이터 조회 로직 (Phase 2 전환 시 사용)

아키텍처 Slide 78-80 기준: Redis 우선 조회 → DynamoDB → Aurora 순서.

```python
# ops-platform/data_service.py
import boto3, json, redis, pymysql
from config import *

def get_dashboard_data(global_id: str) -> dict:
    """Redis → DynamoDB → Aurora 순서로 대시보드 데이터 조회"""

    # 1. Redis 캐시 확인 (TTL 10분, 응답 0.01초)
    r = redis.Redis(host=REDIS_HOST, port=REDIS_PORT, decode_responses=True)
    cache_key = f'dashboard:{global_id}'
    cached = r.get(cache_key)
    if cached:
        return json.loads(cached)

    # 2. DynamoDB에서 점수/등급/AI결과 조회
    dynamo = boto3.resource('dynamodb', region_name=AWS_REGION)
    table = dynamo.Table(DYNAMO_TABLE)
    result = table.get_item(Key={'global_id': global_id})
    item = result.get('Item', {})

    dynamic_score     = float(item.get('dynamic_score', 0))
    grade             = item.get('dynamic_grade', 'BASIC')
    next_best_action  = item.get('next_best_action', '')
    health_score      = float(item.get('health_score', 0))

    # 3. Aurora에서 등급별 추천 상품 조회
    conn = pymysql.connect(host=AURORA_HOST, user=DB_USER,
                           password=DB_PASS, db=DB_NAME)
    with conn.cursor() as cur:
        cur.execute("""
            SELECT p.product_id, p.product_name, p.company_id, p.product_type,
                   GROUP_CONCAT(po.option_value ORDER BY po.sort_order SEPARATOR '||') AS options
            FROM product_master p
            LEFT JOIN product_option po ON p.product_id = po.product_id
            WHERE p.is_active = 1
            GROUP BY p.product_id
            ORDER BY FIELD(p.company_id,'bank','card','insurance','healthcare'), p.product_id
        """)
        products = [
            {'id': r[0], 'name': r[1], 'company': r[2],
             'type': r[3], 'detail': r[4].split('||') if r[4] else []}
            for r in cur.fetchall()
        ]
    conn.close()

    data = {
        'global_id': global_id,
        'dynamic_score': dynamic_score,
        'grade': grade,
        'next_best_action': next_best_action,
        'health_score': health_score,
        'products': products,
    }

    # 4. Redis 캐시 저장 (600초 = 10분)
    r.setex(cache_key, 600, json.dumps(data, ensure_ascii=False))
    return data
```

### DynamoDB 테이블 생성 스크립트

```python
# C:\Users\campus3S026\LS\scripts\create_dynamodb.py
"""
실행: python create_dynamodb.py
사전 조건: AWS CLI 설정 완료, ap-northeast-2 리전
"""
import boto3

def create_table():
    client = boto3.client('dynamodb', region_name='ap-northeast-2')

    try:
        client.create_table(
            TableName='lifesync_customer_result',
            AttributeDefinitions=[
                {'AttributeName': 'global_id', 'AttributeType': 'S'}
            ],
            KeySchema=[
                {'AttributeName': 'global_id', 'KeyType': 'HASH'}
            ],
            BillingMode='PAY_PER_REQUEST',
            TimeToLiveSpecification={    # ttl 필드 기준 자동 삭제
                'AttributeName': 'ttl',
                'Enabled': True
            },
            Tags=[
                {'Key': 'Project', 'Value': 'lifesync360'},
                {'Key': 'Owner',   'Value': 'hwanghc'},
            ]
        )
        print("테이블 생성 완료: lifesync_customer_result")

        # 생성 완료까지 대기
        waiter = client.get_waiter('table_exists')
        waiter.wait(TableName='lifesync_customer_result')
        print("테이블 ACTIVE 상태 확인 완료")

    except client.exceptions.ResourceInUseException:
        print("테이블이 이미 존재합니다.")

if __name__ == '__main__':
```

### 상품 카탈로그 Aurora 시드 스크립트 (JSON 파일 기반)

> **주의**: JSON 파일(`data/product_catalog_seed.json`)에 3가지 문제가 있어 적재 전 확인 필요.  
> 1. `21_고객등급기준추천` 시트 데이터 누락 → `recommend_rule` 테이블 비어있음  
> 2. 보험/카드/예적금/대출 상품의 `description`, `recommendation_condition`, `product_type`이 null → `raw` 안에 데이터는 있음  
> 3. `options`가 운동식단 외 전부 빈 배열 → 보험 보장내용, 카드 혜택 등 미매핑  
> → 지금은 있는 데이터만 적재하고, 데이터셋 담당자에게 JSON 재생성 요청

```python
# C:\Users\campus3S026\LS\scripts\seed_products.py
"""
실행: python seed_products.py
사전 조건: .env 파일 작성 (AURORA_HOST, DB_USER, DB_PASS)
데이터 소스: ../data/product_catalog_seed.json (데이터셋정의서 v9 기반, 148개 상품)
"""
import json, os, sys, pymysql
from pathlib import Path
from dotenv import load_dotenv

load_dotenv()

CATALOG_PATH = Path(__file__).parent.parent / 'data' / 'product_catalog_seed.json'

# ── raw 필드에서 product_type 추출 (시트별 컬럼명이 다름) ──
def extract_product_type(record: dict) -> str:
    raw = record.get('raw', {})
    # 직접 매핑된 경우 우선 사용
    if record.get('product_type'):
        return record['product_type']
    # 시트별 컬럼명 매핑
    for key in ('세부유형', '상품유형', '대출유형', '구분', '투자성향'):
        if key in raw and raw[key]:
            return str(raw[key]).strip()
    return ''

# ── raw 필드에서 description 추출 ──
def extract_description(record: dict) -> str:
    raw = record.get('raw', {})
    if record.get('description'):
        return record['description']
    for key in ('주요 보장 내용', '핵심 혜택', '상품 특징', '기대수익률(연)', '대상 고객 조건'):
        if key in raw and raw[key]:
            return str(raw[key]).strip()
    return ''

# ── raw 필드에서 recommendation_condition 추출 ──
def extract_rec_condition(record: dict) -> str:
    raw = record.get('raw', {})
    if record.get('recommendation_condition'):
        return record['recommendation_condition']
    for key in ('AI 추천 대상', '건강점수 기반 추천 조건', 'AI 추천 타겟', 'AI 추천 트리거', '대상 고객 조건'):
        if key in raw and raw[key]:
            return str(raw[key]).strip()
    return ''

# ── raw 필드에서 options 추출 ──
def extract_options(record: dict) -> list:
    # 이미 options가 있으면 그대로 사용
    if record.get('options'):
        return record['options']

    raw = record.get('raw', {})
    opts = []
    sheet = record.get('source_sheet', '')

    if '보험' in sheet:
        for key in ('주요 보장 내용', '특약 여부', '월 납부액(기준)', '보험기간'):
            if raw.get(key):
                opts.append({'key': key, 'value': str(raw[key])})
    elif '카드' in sheet:
        for key in ('핵심 혜택', '포인트 적립 공식', '연회비(원)', '건강점수 연동 조건'):
            if raw.get(key):
                opts.append({'key': key, 'value': str(raw[key])})
    elif '예적금' in sheet:
        for key in ('기준금리(연)', '우대금리 조건', '가입기간', '최소가입금액'):
            if raw.get(key):
                opts.append({'key': key, 'value': str(raw[key])})
    elif '대출' in sheet:
        for key in ('금리범위', '최대한도', '대출기간', '상환방식'):
            if raw.get(key):
                opts.append({'key': key, 'value': str(raw[key])})
    elif '금융상품' in sheet:
        for key in ('채권/안전자산', '기대수익률(연)', '변동성', '최소 투자금액'):
            if raw.get(key):
                opts.append({'key': key, 'value': str(raw[key])})
    return opts

def main():
    with open(CATALOG_PATH, encoding='utf-8') as f:
        data = json.load(f)

    conn = pymysql.connect(
        host    = os.environ['AURORA_HOST'],
        user    = os.environ['DB_USER'],
        password= os.environ['DB_PASS'],
        db      = 'lifesync',
        charset = 'utf8mb4'
    )

    # ── company_master (JSON company_id 기준 전체 포함) ──
    companies = [
        ('bank',              'LS 은행',         'bank'),
        ('card',              'LS 카드',         'card'),
        ('insurance',         'LS 보험',         'insurance'),
        ('internet_insurance','LS 온라인보험',    'internet_insurance'),
        ('securities',        'LS 증권',         'securities'),
        ('healthcare',        'LS 헬스케어',     'healthcare'),
        ('hospital',          'LS 병원',         'hospital'),
    ]

    inserted_products = 0
    inserted_options  = 0

    with conn.cursor() as cur:
        # 1. company_master
        for cid, cname, ctype in companies:
            cur.execute("""
                INSERT INTO company_master (company_id, company_name, company_type)
                VALUES (%s, %s, %s)
                ON DUPLICATE KEY UPDATE company_name = VALUES(company_name)
            """, (cid, cname, ctype))

        # 2. product_master + product_option
        for catalog in data['catalogs']:
            sheet = catalog['sheet']
            for rec in catalog['records']:
                pid   = rec['product_code']
                cid   = rec['company_id']
                cat   = rec.get('category', '')
                ptype = extract_product_type(rec)
                desc  = extract_description(rec)
                rcond = extract_rec_condition(rec)
                name  = rec['product_name']

                cur.execute("""
                    INSERT INTO product_master
                      (product_id, company_id, category_id, product_type,
                       product_name, product_desc, product_tag, min_grade)
                    VALUES (%s, %s, %s, %s, %s, %s, %s, 'BASIC')
                    ON DUPLICATE KEY UPDATE
                      product_name = VALUES(product_name),
                      product_type = VALUES(product_type),
                      product_desc = VALUES(product_desc)
                """, (pid, cid, cat, ptype, name, desc, ''))
                inserted_products += 1

                # product_option
                opts = extract_options(rec)
                if opts:
                    cur.execute("DELETE FROM product_option WHERE product_id=%s", (pid,))
                    for i, opt in enumerate(opts):
                        cur.execute("""
                            INSERT INTO product_option (product_id, option_key, option_value, sort_order)
                            VALUES (%s, %s, %s, %s)
                        """, (pid, opt.get('key', 'benefit'), opt.get('value', ''), i))
                        inserted_options += 1

                # recommend_rule (recommendation_condition이 있는 경우만)
                if rcond:
                    cur.execute("""
                        INSERT INTO recommend_rule (product_id, condition_key, condition_val, priority)
                        VALUES (%s, 'ai_condition', %s, 0)
                        ON DUPLICATE KEY UPDATE condition_val = VALUES(condition_val)
                    """, (pid, rcond))

    conn.commit()
    conn.close()
    print(f"적재 완료 — 상품: {inserted_products}개 / 옵션: {inserted_options}개")
if __name__ == '__main__':
    main()
```

```python
import os, json, pymysql, boto3
from flask import Flask, render_template, jsonify, request
from functools import wraps

app = Flask(__name__)
ADMIN_TOKEN = os.environ.get('ADMIN_TOKEN', 'admin-dev-token-change-in-prod')

def require_admin(f):
    """간단한 토큰 기반 관리자 인증"""
    @wraps(f)
    def decorated(*args, **kwargs):
        token = request.headers.get('X-Admin-Token', '')
        if token != ADMIN_TOKEN:
            return jsonify({'error': 'unauthorized'}), 401
        return f(*args, **kwargs)
    return decorated

def get_db():

    """일별 신규 가입자 수 (최근 30일)"""
    conn = get_db()
    with conn.cursor() as cur:
        cur.execute("""
            SELECT DATE(created_at) AS dt, COUNT(*) AS cnt
            FROM users
            WHERE created_at >= DATE_SUB(NOW(), INTERVAL 30 DAY)
            GROUP BY DATE(created_at)
            ORDER BY dt
        """)
        data = [{'date': str(r[0]), 'count': r[1]} for r in cur.fetchall()]
    conn.close()
    return jsonify(data)

@app.route('/api/stats/grade')
def api_grade():
    """등급별 분포 (DynamoDB Scan)"""
    table = get_dynamo()
    result = table.scan(ProjectionExpression='dynamic_grade')
    dist = {}
    for item in result['Items']:
        g = item.get('dynamic_grade', 'BASIC')
        dist[g] = dist.get(g, 0) + 1
    return jsonify(dist)

@app.route('/api/stats/recommend')
def api_recommend():
    """추천 상품 클릭률 Top 10"""
    conn = get_db()
    with conn.cursor() as cur:
        cur.execute("""
            SELECT h.product_id, p.product_name,
                   COUNT(*) AS impressions,
                   SUM(h.clicked_at IS NOT NULL) AS clicks,
                   ROUND(SUM(h.clicked_at IS NOT NULL) / COUNT(*) * 100, 1) AS ctr
            FROM customer_recommend_history h
            JOIN product_master p ON h.product_id = p.product_id
            WHERE h.recommended_at >= DATE_SUB(NOW(), INTERVAL 7 DAY)
            GROUP BY h.product_id, p.product_name
            ORDER BY ctr DESC
            LIMIT 10
        """)
        data = [{'product_id': r[0], 'product_name': r[1],
                 'impressions': r[2], 'clicks': r[3], 'ctr': float(r[4] or 0)}
                for r in cur.fetchall()]

    return jsonify(data)
```

## Step 13 | Lambda Recommendation Engine 배포

GCP Cloud Run → AWS API Gateway (Private) → Lambda → DynamoDB/Aurora/Redis 흐름.

### Lambda 코드

```python
# lambda/recommendation_engine.py
import boto3, json, pymysql, os, redis as redis_lib
from decimal import Decimal

dynamo = boto3.resource('dynamodb', region_name='ap-northeast-2')
table = dynamo.Table(os.environ['DYNAMO_TABLE'])

def get_redis():
    return redis_lib.Redis(
        host=os.environ['REDIS_HOST'],
        port=6379,
        decode_responses=True
    )

def get_db():
    return pymysql.connect(
        host     = os.environ['AURORA_HOST'],
        user     = os.environ['DB_USER'],
        password = os.environ['DB_PASS'],
        db       = 'lifesync',
        charset  = 'utf8mb4'
    )

def handler(event, context):
    body = json.loads(event.get('body', '{}'))

    global_id        = body['global_id']
    dynamic_score    = Decimal(str(body['dynamic_score']))
    dynamic_grade    = body['dynamic_grade']
    health_score     = Decimal(str(body['health_score']))
    next_best_action = body.get('next_best_action', '')
    vip_prob         = Decimal(str(body.get('vip_prob', 0)))
    signup_prob      = Decimal(str(body.get('signup_prob', 0)))
    rec_prob         = Decimal(str(body.get('rec_prob', 0)))
    update_time      = body.get('update_time', '')
    ttl              = body.get('ttl', 1777777777)

    # ── 1. DynamoDB 저장 (고객 최신 AI 결과) ──
    table.put_item(Item={
        'global_id':        global_id,
        'update_time':      update_time,
        'dynamic_score':    dynamic_score,
        'dynamic_grade':    dynamic_grade,
        'next_best_action': next_best_action,
        'vip_prob':         vip_prob,
        'signup_prob':      signup_prob,
        'rec_prob':         rec_prob,
        'health_score':     health_score,
        'source':           'GCP_LIFESYNC360',
        'ttl':              ttl,
    })

    # ── 2. Aurora 고객 등급 업데이트 ──
    conn = get_db()
    try:
        with conn.cursor() as cur:
            cur.execute("""
                UPDATE users SET grade=%s WHERE global_id=%s
            """, (dynamic_grade, global_id))

            # 3. Aurora 추천 상품 조회 (등급 기준)
            grade_order = {'PLATINUM': 5, 'GOLD': 4, 'SILVER': 3, 'BRONZE': 2, 'BASIC': 1}
            cur.execute("""
                SELECT p.product_id, p.product_name, p.company_id,
                       p.product_type, p.product_desc, p.product_tag
                FROM product_master p
                WHERE p.is_active = 1
                ORDER BY FIELD(p.company_id,'bank','card','insurance','healthcare'),
                         p.product_id
            """)
            all_products = cur.fetchall()
            products = [
                {'id': r[0], 'name': r[1], 'company': r[2],
                 'type': r[3], 'desc': r[4], 'tag': r[5]}
                for r in all_products
            ]

            # 추천 이력 기록
            for p in products[:3]:   # 상위 3개만 기록
                cur.execute("""
                    INSERT INTO customer_recommend_history (global_id, product_id)
                    VALUES (%s, %s)
                """, (global_id, p['id']))

        conn.commit()
    finally:
        conn.close()

    # ── 4. Redis 캐시 갱신 (TTL 10분) ──
    r = get_redis()
    cache_data = {
        'global_id':        global_id,
        'grade':            dynamic_grade,
        'score':            float(dynamic_score),
        'health_score':     float(health_score),
        'next_best_action': next_best_action,
        'products':         products,
        'updated_at':       update_time,
    }
    r.setex(f'dashboard:{global_id}', 600,
            json.dumps(cache_data, ensure_ascii=False))

    return {
        'statusCode': 200,
        'body': json.dumps({'status': 'ok', 'global_id': global_id})
    }
```
